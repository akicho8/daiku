: quick_table { :a }

: quick_table { [:a, 1] }

:

: 
: quick_table { [[:a, 1], [:b, 2]] }
:   [a][1]
:   [b][2]
: 
: quick_table { {:a => 1, :b => 2} }
:   [a][1]
:   [b][2]
: 
: quick_table(:array_of_hash => true) { [{:a => 1, :b => 2}, {:a => 3, :b => 4}] }

:   [1][2]
